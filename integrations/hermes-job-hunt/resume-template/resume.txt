/* =========================================================
     Resume Tweaks App
     Three expressive controls that reshape feel, not pixels:
       • accent   — swap the signature gold for another restrained ink
       • voice    — Restrained / Editorial / Quiet (italic register)
       • density  — Open / Standard / Tight (rhythm)
     ========================================================= */

  // Accent palette: each option = [gold, gold-soft]. gold-soft is the
  // 18%-alpha wash for company-name backgrounds.
  const ACCENTS = {
    gold:  { gold: '#FFB703', soft: 'rgba(255, 183, 3, 0.18)' },
    coral: { gold: '#EB4D56', soft: 'rgba(235, 77, 86, 0.16)' },
    teal:  { gold: '#12A89C', soft: 'rgba(18, 168, 156, 0.18)' },
    mono:  { gold: '#093444', soft: 'rgba(9, 52, 68, 0.10)' },
  };
  const ACCENT_LABELS = { gold: 'Gold', coral: 'Coral', teal: 'Teal', mono: 'Mono' };

  function applyAccent(key) {
    const a = ACCENTS[key] || ACCENTS.gold;
    document.documentElement.style.setProperty('--gold', a.gold);
    document.documentElement.style.setProperty('--gold-soft', a.soft);
  }

  function applyVoice(value) {
    const b = document.body;
    b.classList.remove('voice-restrained', 'voice-editorial', 'voice-quiet');
    b.classList.add('voice-' + value);
  }

  function applyDensity(value) {
    const b = document.body;
    b.classList.remove('density-open', 'density-standard', 'density-tight');
    b.classList.add('density-' + value);
  }

  function ResumeTweaks() {
    const [t, setTweak] = useTweaks(window.TWEAK_DEFAULTS);

    // Live-apply on every change
    React.useEffect(() => { applyAccent(t.accent); },   [t.accent]);
    React.useEffect(() => { applyVoice(t.voice); },     [t.voice]);
    React.useEffect(() => { applyDensity(t.density); }, [t.density]);

    return (
      <TweaksPanel title="Tweaks">
        <TweakSection label="Signature accent" />
        <TweakColor
          label="Accent"
          value={ACCENTS[t.accent].gold}
          options={Object.values(ACCENTS).map(a => a.gold)}
          onChange={(hex) => {
            const k = Object.keys(ACCENTS).find(k => ACCENTS[k].gold === hex) || 'gold';
            setTweak('accent', k);
          }}
        />
        <div style={{
          fontSize: 10.5, color: 'rgba(41,38,27,.55)',
          marginTop: -2, lineHeight: 1.35,
        }}>
          {ACCENT_LABELS[t.accent]} — replaces the signature gold across
          rules, bullets, dots, and folios.
        </div>

        <TweakSection label="Editorial voice" />
        <TweakRadio
          label="Voice"
          value={t.voice}
          options={['restrained', 'editorial', 'quiet']}
          onChange={(v) => setTweak('voice', v)}
        />
        <div style={{
          fontSize: 10.5, color: 'rgba(41,38,27,.55)',
          marginTop: -2, lineHeight: 1.35,
        }}>
          {t.voice === 'restrained' && 'Italic Fraunces shows up only in the tagline, opener, and closing line.'}
          {t.voice === 'editorial'  && 'Section labels, role titles, and dates lean into Fraunces italic — magazine spread.'}
          {t.voice === 'quiet'      && 'All roman, no italic. Reads as a clean serious technical doc.'}
        </div>

        <TweakSection label="Rhythm" />
        <TweakRadio
          label="Density"
          value={t.density}
          options={['open', 'standard', 'tight']}
          onChange={(v) => setTweak('density', v)}
        />
        <div style={{
          fontSize: 10.5, color: 'rgba(41,38,27,.55)',
          marginTop: -2, lineHeight: 1.35,
        }}>
          {t.density === 'open'     && 'More breathing room. Expect 4 pages on the master superset.'}
          {t.density === 'standard' && 'Default rhythm — 3-page master / 2-page per-job target.'}
          {t.density === 'tight'    && 'Compressed line-heights and margins. Pushes toward 2 pages.'}
        </div>
      </TweaksPanel>
    );
  }

  // Apply persisted defaults before mounting so the page looks "right"
  // even if the user never opens the panel.
  applyAccent(window.TWEAK_DEFAULTS.accent);
  applyVoice(window.TWEAK_DEFAULTS.voice);
  applyDensity(window.TWEAK_DEFAULTS.density);

  const root = ReactDOM.createRoot(document.getElementById('tweaks-root'));
  root.render(<ResumeTweaks />);
